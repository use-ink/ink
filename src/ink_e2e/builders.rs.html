15use ink_env::{
16    call::{
17        utils::{
18            ReturnType,
19            Set,
20        },
21        CreateBuilder,
22        ExecutionInput,
23        LimitParamsV2,
24    },
25    Environment,
26};
27use ink_primitives::abi::AbiEncodeWith;
28
29use crate::H256;
30
31/// The type returned from `ContractRef` constructors, partially initialized with the
32/// execution input arguments.
33pub type CreateBuilderPartial<E, ContractRef, Args, R, Abi> = CreateBuilder<
34    E,
35    ContractRef,
36    Set<LimitParamsV2>,
37    Set<ExecutionInput<Args, Abi>>,
38    Set<ReturnType<R>>,
39    Abi,
40>;
41
42/// Get the encoded constructor arguments from the partially initialized `CreateBuilder`
43pub fn constructor_exec_input<E, ContractRef, Args: AbiEncodeWith<Abi>, R, Abi>(
44    builder: CreateBuilderPartial<E, ContractRef, Args, R, Abi>,
45) -> Vec<u8>
46where
47    E: Environment,
48{
49    // set all the other properties to default values, we only require the `exec_input`.
50    builder
51        .endowment(0u32.into())
52        .code_hash(H256::zero())
53        .salt_bytes(None)
54        .params()
55        .exec_input()
56        .encode()
57}
