55/// Environmental contract functionality that does not require `Environment`.
56pub trait EnvBackend {
57    /// Writes the value to the contract storage under the given storage key.
58    ///
59    /// Returns the size of the pre-existing value at the specified key if any.
60    fn set_contract_storage<K, V>(&mut self, key: &K, value: &V) -> Option<u32>
61    where
62        K: scale::Encode,
63        V: Storable;
64
65    /// Returns the value stored under the given storage key in the contract's storage if
66    /// any.
67    ///
68    /// # Errors
69    ///
70    /// - If the decoding of the typed value failed
71    fn get_contract_storage<K, R>(&mut self, key: &K) -> Result<Option<R>>
72    where
73        K: scale::Encode,
74        R: Storable;
75
76    /// Removes the `value` at `key`, returning the previous `value` at `key` from storage
77    /// if any.
78    ///
79    /// # Errors
80    ///
81    /// - If the decoding of the typed value failed
82    fn take_contract_storage<K, R>(&mut self, key: &K) -> Result<Option<R>>
83    where
84        K: scale::Encode,
85        R: Storable;
86
87    /// Returns the size of a value stored under the given storage key is returned if any.
88    fn contains_contract_storage<K>(&mut self, key: &K) -> Option<u32>
89    where
90        K: scale::Encode;
91
92    /// Clears the contract's storage key entry under the given storage key.
93    ///
94    /// Returns the size of the previously stored value at the specified key if any.
95    fn clear_contract_storage<K>(&mut self, key: &K) -> Option<u32>
96    where
97        K: scale::Encode;
98
99    /// Returns the execution input to the executed contract and decodes it as `T`.
100    ///
101    /// # Note
102    ///
103    /// - The input is the 4-bytes selector followed by the arguments of the called
104    ///   function in their SCALE encoded representation.
105    /// - No prior interaction with the environment must take place before calling this
106    ///   procedure.
107    ///
108    /// # Usage
109    ///
110    /// Normally contracts define their own `enum` dispatch types respective
111    /// to their exported constructors and messages that implement `scale::Decode`
112    /// according to the constructors or messages selectors and their arguments.
113    /// These `enum` dispatch types are then given to this procedure as the `T`.
114    ///
115    /// When using ink! users do not have to construct those enum dispatch types
116    /// themselves as they are normally generated by the ink! code generation
117    /// automatically.
118    ///
119    /// # Errors
120    ///
121    /// If the given `T` cannot be properly decoded from the expected input.
122    fn decode_input<T>(&mut self) -> core::result::Result<T, DispatchError>
123    where
124        T: DecodeDispatch;
125
126    /// Returns the value back to the caller of the executed contract.
127    ///
128    /// # Note
129    ///
130    /// Calling this method will end contract execution immediately.
131    /// It will return the given return value back to its caller.
132    ///
133    /// The `flags` parameter can be used to revert the state changes of the
134    /// entire execution if necessary.
135    #[cfg(not(feature = "std"))]
136    fn return_value<R>(&mut self, flags: ReturnFlags, return_value: &R) -> !
137    where
138        R: scale::Encode;
139
140    /// Returns the value back to the caller of the executed contract.
141    ///
142    /// # Note
143    ///
144    /// When the `std` feature is used, the contract is allowed to
145    /// return normally. This feature should only be used for integration tests.
146    ///
147    /// The `flags` parameter can be used to revert the state changes of the
148    /// entire execution if necessary.
149    #[cfg(feature = "std")]
150    fn return_value<R>(&mut self, flags: ReturnFlags, return_value: &R)
151    where
152        R: scale::Encode;
153
154    /// Returns the *Solidity ABI encoded* value back to the caller of the executed
155    /// contract.
156    ///
157    /// # Note
158    ///
159    /// This function stops the execution of the contract immediately.
160    fn return_value_solidity<R>(&mut self, flags: ReturnFlags, return_value: &R) -> !
161    where
162        R: for<'a> SolResultEncode<'a>;
163
164    /// Conducts the crypto hash of the given input and stores the result in `output`.
165    fn hash_bytes<H>(&mut self, input: &[u8], output: &mut <H as HashOutput>::Type)
166    where
167        H: CryptoHash;
168
169    /// Conducts the crypto hash of the given encoded input and stores the result in
170    /// `output`.
171    fn hash_encoded<H, T>(&mut self, input: &T, output: &mut <H as HashOutput>::Type)
172    where
173        H: CryptoHash,
174        T: scale::Encode;
175
176    /// Calls the `bn128_add` G1 addition precompile.
177    ///
178    /// Inputs are affine G1 coordinates over Fq.
179    /// Returns the resulting affine point or (0, 0) if the result is ∞ / invalid.
180    ///
181    /// # Note
182    ///
183    /// The precompile address is `0x06`. You can find its implementation here:
184    /// <https://github.com/paritytech/polkadot-sdk/blob/master/substrate/frame/revive/src/precompiles/builtin/bn128.rs>
185    fn bn128_add(&mut self, x1: U256, y1: U256, x2: U256, y2: U256) -> (U256, U256);
186
187    /// Calls the `bn128_mul` G1 scalar-mul precompile.
188    ///
189    /// Multiplies an affine G1 point (x1, y1) by a scalar ∈ Fr.
190    /// Returns the resulting affine point or (0, 0) if the result is ∞ / invalid.
191    ///
192    /// # Note
193    ///
194    /// The precompile address is `0x07`. You can find its implementation here:
195    /// <https://github.com/paritytech/polkadot-sdk/blob/master/substrate/frame/revive/src/precompiles/builtin/bn128.rs>
196    fn bn128_mul(&mut self, x1: U256, y1: U256, scalar: U256) -> (U256, U256);
197
198    /// Calls the `bn128_pairing` precompile.
199    ///
200    /// Input is the Solidity-ABI-encoded sequence of (G1, G2) pairs.
201    /// Returns `true` iff the product of pairings evaluates to the identity.
202    ///
203    /// # Note
204    ///
205    /// The precompile address is `0x08`. You can find its implementation here:
206    /// <https://github.com/paritytech/polkadot-sdk/blob/master/substrate/frame/revive/src/precompiles/builtin/bn128.rs>
207    fn bn128_pairing(&mut self, input: &[u8]) -> bool;
208
209    /// Recovers the compressed ECDSA public key for given `signature` and `message_hash`,
210    /// and stores the result in `output`.
211    fn ecdsa_recover(
212        &mut self,
213        signature: &[u8; 65],
214        message_hash: &[u8; 32],
215        output: &mut [u8; 33],
216    ) -> Result<()>;
217
218    /// Retrieves an Ethereum address from the ECDSA compressed `pubkey`
219    /// and stores the result in `output`.
220    #[cfg(feature = "unstable-hostfn")]
221    fn ecdsa_to_eth_address(
222        &mut self,
223        pubkey: &[u8; 33],
224        output: &mut [u8; 20],
225    ) -> Result<()>;
226
227    /// Verifies a sr25519 signature.
228    ///
229    /// # Errors
230    ///
231    /// - If the signature verification failed.
232    ///
233    /// **WARNING**: this function is from the [unstable interface](https://github.com/paritytech/polkadot-sdk/tree/master/substrate/frame/revive#unstable-interfaces)
234    /// which is unsafe and normally is not available on production chains.
235    #[cfg(feature = "unstable-hostfn")]
236    fn sr25519_verify(
237        &mut self,
238        signature: &[u8; 64],
239        message: &[u8],
240        pub_key: &[u8; 32],
241    ) -> Result<()>;
242
243    /// Sets a new code hash for the current contract.
244    ///
245    /// This effectively replaces the code which is executed for this contract address.
246    ///
247    /// # Errors
248    ///
249    /// - If the supplied `code_hash` cannot be found on-chain.
250    #[cfg(feature = "unstable-hostfn")]
251    fn set_code_hash(&mut self, code_hash: &H256) -> Result<()>;
252
253    /// Returns the size of the buffer that is remaining in the backend.
254    fn remaining_buffer(&mut self) -> usize;
255}

257/// Environmental contract functionality.
258pub trait TypedEnvBackend: EnvBackend {
259    /// Returns the address of the caller of the executed contract.
260    ///
261    /// # Note
262    ///
263    /// For more details visit: [`caller`][`crate::caller`]
264    fn caller(&mut self) -> Address;
265
266    /// Returns the block's `ref_time` limit.
267    /// This is akin to the EVM [GASLIMIT](https://www.evm.codes/?fork=cancun#45) opcode.
268    ///
269    /// # Note
270    ///
271    /// For more details visit: [`gas_limit`][`crate::gas_limit`]
272    fn gas_limit(&mut self) -> u64;
273
274    /// Returns the price per `ref_time`, akin to the EVM
275    /// [GASPRICE](https://www.evm.codes/?fork=cancun#3a) opcode.
276    ///
277    /// See <https://use.ink/docs/v6/basics/gas/#what-is-gas-in-ink> for more information.
278    ///
279    /// # Note
280    ///
281    /// For more details visit: [`gas_price`][`crate::gas_price`]
282    fn gas_price(&mut self) -> u64;
283
284    /// Returns the amount of gas left.
285    /// This is the `ref_time` left.
286    ///
287    /// See <https://use.ink/docs/v6/basics/gas/#what-is-gas-in-ink> for more information.
288    ///
289    /// # Note
290    ///
291    /// For more details visit: [`gas_left`][`crate::gas_left`]
292    fn gas_left(&mut self) -> u64;
293
294    /// Returns the total size of the contract call input data.
295    /// This is akin to the EVM [CALLDATASIZE](https://www.evm.codes/?fork=cancun#36) opcode.
296    ///
297    /// # Note
298    ///
299    /// For more details visit: [`call_data_size`][`crate::call_data_size]
300    fn call_data_size(&mut self) -> u64;
301
302    /// Returns the size of the returned data of the last contract call or instantiation.
303    /// This is akin to the EVM [RETURNDATASIZE](https://www.evm.codes/?fork=cancun#3d) opcode.
304    ///
305    /// # Note
306    ///
307    /// For more details visit: [`return_data_size`][`crate::return_data_size]
308    fn return_data_size(&mut self) -> u64;
309
310    /// Returns the transferred value for the contract execution.
311    ///
312    /// # Note
313    ///
314    /// For more details visit: [`transferred_value`][`crate::transferred_value`]
315    fn transferred_value(&mut self) -> U256;
316
317    /// Returns the price for the specified amount of gas.
318    ///
319    /// # Note
320    ///
321    /// For more details visit: [`weight_to_fee`][`crate::weight_to_fee`]
322    fn weight_to_fee(&mut self, gas: u64) -> U256;
323
324    /// Returns the timestamp of the current block.
325    ///
326    /// # Note
327    ///
328    /// For more details visit: [`block_timestamp`][`crate::block_timestamp`]
329    fn block_timestamp<E: Environment>(&mut self) -> E::Timestamp;
330
331    /// Retrieves the account id for a specified address.
332    ///
333    /// # Note
334    ///
335    /// For more details visit: [`to_account_id`][`crate::to_account_id`]
336    fn to_account_id<E: Environment>(&mut self, addr: Address) -> E::AccountId;
337
338    /// Returns the address of the executed contract.
339    ///
340    /// # Note
341    ///
342    /// For more details visit: [`account_id`][`crate::account_id`]
343    fn account_id<E: Environment>(&mut self) -> E::AccountId;
344
345    /// Returns the address of the executed contract.
346    ///
347    /// # Note
348    ///
349    /// For more details visit: [`address`][`crate::address`]
350    fn address(&mut self) -> Address;
351
352    /// Returns the balance of the executed contract.
353    ///
354    /// # Note
355    ///
356    /// For more details visit: [`balance`][`crate::balance`]
357    fn balance(&mut self) -> U256;
358
359    /// Returns the current block number.
360    ///
361    /// # Note
362    ///
363    /// For more details visit: [`block_number`][`crate::block_number`]
364    fn block_number<E: Environment>(&mut self) -> E::BlockNumber;
365
366    /// Returns the minimum balance that is required for creating an account
367    /// (i.e. the chain's existential deposit).
368    ///
369    /// # Note
370    ///
371    /// For more details visit: [`minimum_balance`][`crate::minimum_balance`]
372    fn minimum_balance(&mut self) -> U256;
373
374    /// Emits an event with the given event data.
375    ///
376    /// # Note
377    ///
378    /// For more details visit: [`emit_event`][`crate::emit_event`]
379    fn emit_event<Evt, Abi>(&mut self, event: &Evt)
380    where
381        Evt: Event<Abi>,
382        Abi: TopicEncoder;
383
384    /// Invokes a contract message and returns its result.
385    ///
386    /// # Note
387    ///
388    /// **This will call into the latest `call_v2` host function.**
389    ///
390    /// For more details visit: [`invoke_contract`][`crate::invoke_contract`]
391    fn invoke_contract<E, Args, R, Abi>(
392        &mut self,
393        call_data: &CallParams<E, Call, Args, R, Abi>,
394    ) -> Result<ink_primitives::MessageResult<R>>
395    where
396        E: Environment,
397        Args: EncodeArgsWith<Abi>,
398        R: DecodeMessageResult<Abi>;
399
400    /// Invokes a contract message via delegate call and returns its result.
401    ///
402    /// # Note
403    ///
404    /// For more details visit:
405    /// [`invoke_contract_delegate`][`crate::invoke_contract_delegate`]
406    fn invoke_contract_delegate<E, Args, R, Abi>(
407        &mut self,
408        call_data: &CallParams<E, DelegateCall, Args, R, Abi>,
409    ) -> Result<ink_primitives::MessageResult<R>>
410    where
411        E: Environment,
412        Args: EncodeArgsWith<Abi>,
413        R: DecodeMessageResult<Abi>;
414
415    /// Instantiates another contract.
416    ///
417    /// # Note
418    ///
419    /// For more details visit: [`instantiate_contract`][`crate::instantiate_contract`]
420    fn instantiate_contract<E, ContractRef, Args, R, Abi>(
421        &mut self,
422        params: &CreateParams<E, ContractRef, LimitParamsV2, Args, R, Abi>,
423    ) -> Result<
424        ink_primitives::ConstructorResult<
425            <R as ConstructorReturnType<ContractRef, Abi>>::Output,
426        >,
427    >
428    where
429        E: Environment,
430        ContractRef: FromAddr + crate::ContractReverseReference,
431        <ContractRef as crate::ContractReverseReference>::Type:
432            crate::reflect::ContractConstructorDecoder,
433        Args: EncodeArgsWith<Abi>,
434        R: ConstructorReturnType<ContractRef, Abi>;
435
436    /// Terminates a smart contract.
437    ///
438    /// # Note
439    ///
440    /// For more details visit: [`terminate_contract`][`crate::terminate_contract`]
441    #[cfg(feature = "unstable-hostfn")]
442    fn terminate_contract(&mut self, beneficiary: Address) -> !;
443
444    /// Transfers value from the contract to the destination account ID.
445    ///
446    /// # Note
447    ///
448    /// For more details visit: [`transfer`][`crate::transfer`]
449    fn transfer<E>(&mut self, destination: Address, value: U256) -> Result<()>
450    where
451        E: Environment;
452
453    /// Checks whether a specified contract lives at `addr`.
454    ///
455    /// # Note
456    ///
457    /// For more details visit: [`is_contract`][`crate::is_contract`]
458    fn is_contract(&mut self, account: &Address) -> bool;
459
460    /// Checks whether the caller of the current contract is the origin of the whole call
461    /// stack.
462    ///
463    /// # Note
464    ///
465    /// For more details visit: [`caller_is_origin`][`crate::caller_is_origin`]
466    fn caller_is_origin(&mut self) -> bool;
467
468    /// Checks whether the caller of the current contract is root.
469    ///
470    /// # Note
471    ///
472    /// For more details visit: [`caller_is_root`][`crate::caller_is_root`]
473    fn caller_is_root(&mut self) -> bool;
474
475    /// Retrieves the code hash of the contract at the given `account` id.
476    ///
477    /// # Note
478    ///
479    /// For more details visit: [`code_hash`][`crate::code_hash`]
480    fn code_hash(&mut self, account: &Address)
481    -> core::result::Result<H256, CodeHashErr>;
482
483    /// Retrieves the code hash of the currently executing contract.
484    ///
485    /// # Note
486    ///
487    /// For more details visit: [`own_code_hash`][`crate::own_code_hash`]
488    fn own_code_hash(&mut self) -> H256;
489
490    /// Estimates the [`Weight`] required to execute a given XCM message.
491    ///
492    /// # Note
493    ///
494    /// For more details visit: [`xcm`][`crate::xcm_weigh`].
495    #[cfg(feature = "xcm")]
496    fn xcm_weigh<Call>(&mut self, msg: &xcm::VersionedXcm<Call>) -> Result<Weight>
497    where
498        Call: scale::Encode;
499
500    /// Execute an XCM message locally, using the contract's address as the origin.
501    ///
502    /// # Note
503    ///
504    /// For more details visit: [`xcm`][`crate::xcm_execute`].
505    #[cfg(feature = "xcm")]
506    fn xcm_execute<Call>(
507        &mut self,
508        msg: &xcm::VersionedXcm<Call>,
509        weight: Weight,
510    ) -> Result<()>
511    where
512        Call: scale::Encode;
513
514    /// Send an XCM message, using the contract's address as the origin.
515    ///
516    /// # Note
517    ///
518    /// For more details visit: [`xcm`][`crate::xcm_send`].
519    #[cfg(feature = "xcm")]
520    fn xcm_send<Call>(
521        &mut self,
522        dest: &xcm::VersionedLocation,
523        msg: &xcm::VersionedXcm<Call>,
524    ) -> Result<()>
525    where
526        Call: scale::Encode;
