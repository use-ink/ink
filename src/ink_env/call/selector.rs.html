15use derive_more::From;
16
17/// The function selector.
18#[derive(
19    Default, Debug, Copy, Clone, PartialEq, Eq, From, scale::Decode, scale::Encode,
20)]
21pub struct Selector {
22    /// The 4 underlying bytes.
23    bytes: [u8; 4],
24}
25
26impl Selector {
27    /// Creates a selector directly from 4 bytes.
28    pub const fn new(bytes: [u8; 4]) -> Self {
29        Self { bytes }
30    }
31
32    /// Returns the underlying bytes of the selector.
33    pub const fn to_bytes(self) -> [u8; 4] {
34        self.bytes
35    }
36}
