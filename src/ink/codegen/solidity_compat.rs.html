15use keccak_const::Keccak256;
16
17/// Compile-time `format!`-like macro that returns `&'static str`.
18///
19/// The first argument is a format string which can be a `String` (i.e. doesn't have
20/// to be a literal). Like `format!` the format string can contain `{}`s which are
21/// replace by the additional parameters which must all be constant expressions.
22///
23/// See [`const_format::formatc`] for details.
24pub use const_format::formatc as const_format;
25
26/// Compile-time Solidity selector computation.
27///
28/// Takes function signature as a string and returns a 4 byte representation of the
29/// selector.
30pub const fn sol_selector_bytes(sig: &str) -> [u8; 4] {
31    let hash = Keccak256::new().update(sig.as_bytes()).finalize();
32    [hash[0], hash[1], hash[2], hash[3]]
33}
