15/// Takes a generic type as input and just consumes it while doing nothing.
16///
17/// # Note
18///
19/// This can be used to trigger some compile time checks due to the fact
20/// that the type consumed this way is type checked. We usually use this
21/// to make the Rust compiler check the trait bounds in particular.
22///
23/// # Usage: Compiles
24///
25/// ```
26/// # use ink::codegen::utils::consume_type;
27/// # use core::marker::PhantomData;
28/// #
29/// pub struct RequiresCopy<T: Copy>(PhantomData<T>);
30///
31/// // The following line of code works because `i32: Copy`.
32/// consume_type::<RequiresCopy<i32>>();
33/// ```
34///
35/// # Usage: Compile Error
36///
37/// ```compile_fail
38/// # use ink::codegen::utils::consume_type;
39/// # use core::marker::PhantomData;
40/// #
41/// pub struct RequiresCopy<T: Copy>(PhantomData<T>);
42///
43/// // The following line of code fails to compile because
44/// // `String` does not implement `Copy`.
45/// consume_type::<RequiresCopy<String>>();
46/// ```
47pub const fn consume_type<T>() {}
