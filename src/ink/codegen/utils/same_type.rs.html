15use core::marker::PhantomData;
16
17/// Can be used to check equality of types.
18///
19/// # Example
20///
21/// This code compiles:
22///
23/// ```
24/// # use ink::codegen::utils::IsSameType;
25/// const _: IsSameType<i32> = IsSameType::<i32>::new();
26/// ```
27///
28/// While this code does not:
29///
30/// ```compile_fail
31/// # use ink::codegen::utils::IsSameType;
32/// const _: IsSameType<i32> = IsSameType::<i64>::new();
33/// ```
34pub struct IsSameType<T> {
35    _marker: PhantomData<T>,
36}
37
38impl<T> IsSameType<T> {
39    /// Creates a new const instance.
40    pub const fn new() -> Self {
41        Self {
42            _marker: PhantomData,
43        }
44    }
45}
46
47impl<T> Default for IsSameType<T> {
48    fn default() -> Self {
49        Self::new()
50    }
51}
