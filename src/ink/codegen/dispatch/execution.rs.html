15use ink_env::DispatchError;
16
17/// Returns `Ok` if the caller did not transfer additional value to the callee.
18///
19/// # Errors
20///
21/// If the caller did send some amount of transferred value to the callee.
22#[inline]
23pub fn deny_payment() -> Result<(), DispatchError> {
24    let transferred = ink_env::transferred_value();
25    if !transferred.is_zero() {
26        return Err(DispatchError::PaidUnpayableMessage)
27    }
28    Ok(())
29}
