28/// Trait used as bound for the outputs of ink! trait definition messages.
29///
30/// # Note
31///
32/// Indicates that a value of type `T` or an equivalent `CallBuilder`
33/// instantiation of an ink! message returning a value of type `T` are
34/// returned by the respective ink! trait message.
35///
36/// The bounds are automatically generated by the `#[ink::trait_definition]`
37/// procedural macro.
38pub trait ImpliesReturn<T> {}
39
40impl<T> ImpliesReturn<T> for T {}
41impl<T, E, CallType, Args, Abi> ImpliesReturn<T>
42    for CallBuilder<E, Set<CallType>, Set<ExecutionInput<Args, Abi>>, Set<ReturnType<T>>>
43where
44    E: Environment,
45{
46}
47
48impl<E, CallType, Args, Abi> ImpliesReturn<()>
49    for CallBuilder<E, Set<CallType>, Set<ExecutionInput<Args, Abi>>, Set<()>>
50where
51    E: Environment,
52{
53}
54
55impl<Args, RetType, Abi> ImpliesReturn<RetType> for Execution<Args, RetType, Abi> {}
