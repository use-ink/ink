17/// Generates a wrapper which can be used for interacting with the contract.
18///
19/// This is done by creating a wrapper around the trait defined with the
20/// [`ink::trait_definition`](crate::trait_definition) macro.
21///
22/// The macro returns the native Rust type that implements the corresponding trait,
23/// so it can be used in any Rust context that expects types.
24///
25/// # Usage in the `#[ink::contract]` context
26///
27/// The macro expects one argument â€’ the path to the trait, in order to create a wrapper
28/// around it.
29///
30/// ```rust
31/// #[ink::contract]
32/// mod trait_caller {
33///     use ink::{
34///         contract_ref,
35///         H160,
36///         U256,
37///     };
38///
39///     #[ink::trait_definition]
40///     pub trait Erc20 {
41///         /// Returns the total supply of the ERC-20 smart contract.
42///         #[ink(message)]
43///         fn total_supply(&self) -> U256;
44///
45///         /// Transfers balance from the caller to the given address.
46///         #[ink(message)]
47///         fn transfer(&mut self, amount: U256, to: H160) -> bool;
48///     }
49///
50///     #[ink(storage)]
51///     pub struct Caller {
52///         /// The example of `contract_ref!` as a struct type.
53///         erc20: contract_ref!(Erc20),
54///     }
55///
56///     impl Caller {
57///         /// Example of `contract_ref!` as an argument type.
58///         #[ink(constructor)]
59///         pub fn new(erc20: contract_ref!(Erc20)) -> Self {
60///             Self { erc20 }
61///         }
62///
63///         /// Example of converting `AccountId` into `contract_ref!` implicitly.
64///         #[ink(message)]
65///         pub fn change_account_id_1(&mut self, new_erc20: H160) {
66///             self.erc20 = new_erc20.into();
67///         }
68///
69///         /// Example of converting `AccountId` into `contract_ref!` explicitly.
70///         #[ink(message)]
71///         pub fn change_account_id_2(&mut self, new_erc20: H160) {
72///             let erc20: contract_ref!(Erc20) = new_erc20.into();
73///             self.erc20 = erc20;
74///         }
75///
76///         /// Example of converting `AccountId` into an alias from `contract_ref!`.
77///         #[ink(message)]
78///         pub fn change_account_id_3(&mut self, new_erc20: H160) {
79///             type Erc20Wrapper = contract_ref!(Erc20);
80///             let erc20: Erc20Wrapper = new_erc20.into();
81///             self.erc20 = erc20;
82///         }
83///
84///         /// Example of how to do common calls via fully qualified syntax.
85///         #[ink(message)]
86///         pub fn total_supply_1(&self) -> U256 {
87///             Erc20::total_supply(&self.erc20)
88///         }
89///
90///         /// Example of how to do common calls without fully qualified syntax.
91///         #[ink(message)]
92///         pub fn total_supply_2(&self) -> U256 {
93///             self.erc20.total_supply()
94///         }
95///
96///         /// Example of how to use the call builder with `contract_ref!`.
97///         #[ink(message)]
98///         pub fn total_supply_3(&self) -> U256 {
99///             use ink::codegen::TraitCallBuilder;
100///             // Returns the `CallBuilder` that implements `Erc20` trait.
101///             let erc20_builder = self.erc20.call();
102///             erc20_builder
103///                 .total_supply()
104///                 .transferred_value(U256::from(0))
105///                 .invoke()
106///         }
107///
108///         /// Example of how to do common calls and convert
109///         /// the `contract_ref!` into `AccountId`.
110///         #[ink(message)]
111///         pub fn transfer_to_erc20(&mut self, amount: U256) -> bool {
112///             let erc20_as_account_id = self.erc20.as_ref().clone();
113///             self.erc20.transfer(amount, erc20_as_account_id)
114///         }
115///     }
116/// }
117/// ```
118///
119/// # Usage outside the `#[ink::contract]` context
120///
121/// The macro expects two arguments:
122/// - The first argument is the path to the trait, e.g. `Erc20` or `erc20::Erc20`.
123/// - The second argument is the type of the [`ink_env::Environment`].
124///
125/// If the second argument is not specified, the macro uses the `Environment` type alias.
126///
127/// ```rust
128/// use ink::contract_ref;
129/// use ink_env::DefaultEnvironment;
130/// use ink_primitives::H160;
131///
132/// #[ink::trait_definition]
133/// pub trait Erc20 {
134///     /// Returns the total supply of the ERC-20 smart contract.
135///     #[ink(message)]
136///     fn total_supply(&self) -> u128;
137///
138///     /// Transfers balance from the caller to the given address.
139///     #[ink(message)]
140///     fn transfer(&mut self, amount: u128, to: H160) -> bool;
141/// }
142///
143/// #[derive(Clone)]
144/// pub struct CustomEnv;
145///
146/// impl ink_env::Environment for CustomEnv {
147///     const MAX_EVENT_TOPICS: usize = 3;
148///     type AccountId = [u8; 32];
149///     type Balance = u64;
150///     type Hash = [u8; 32];
151///     type Timestamp = u64;
152///     type BlockNumber = u64;
153///     type ChainExtension = ();
154///     type EventRecord = ();
155/// }
156///
157/// type AliasWithDefaultEnv = contract_ref!(Erc20, DefaultEnvironment);
158/// type AliasWithCustomEnv = contract_ref!(Erc20, CustomEnv);
159/// type AliasWithGenericEnv<E> = contract_ref!(Erc20, E);
160///
161/// fn default(mut contract: contract_ref!(Erc20, DefaultEnvironment)) {
162///     let total_supply = contract.total_supply();
163///     let to: H160 = contract.as_ref().clone();
164///     contract.transfer(total_supply, to);
165/// }
166///
167/// fn default_alias(mut contract: AliasWithDefaultEnv) {
168///     default(contract)
169/// }
170///
171/// fn custom(mut contract: contract_ref!(Erc20, CustomEnv)) {
172///     let total_supply = contract.total_supply();
173///     contract.transfer(total_supply, contract.as_ref().clone());
174/// }
175///
176/// fn custom_alias(mut contract: AliasWithCustomEnv) {
177///     custom(contract)
178/// }
179///
180/// fn generic<E, A>(mut contract: contract_ref!(Erc20, E))
181/// where
182///     E: ink_env::Environment<AccountId = A>,
183///     A: Into<H160> + Clone,
184/// {
185///     let total_supply = contract.total_supply();
186///     let to = contract.as_ref().clone();
187///     contract.transfer(total_supply, to.into());
188/// }
189///
190/// fn generic_alias<E, A>(mut contract: AliasWithGenericEnv<E>)
191/// where
192///     E: ink_env::Environment<AccountId = A>,
193///     A: Into<H160> + Clone,
194/// {
195///     generic(contract)
196/// }
197///
198/// type Environment = DefaultEnvironment;
199///
200/// fn contract_ref_default_behaviour(mut contract: contract_ref!(Erc20)) {
201///     let total_supply = contract.total_supply();
202///     let to: H160 = contract.as_ref().clone();
203///     contract.transfer(total_supply, to);
204/// }
205/// ```
206#[macro_export]
207macro_rules! contract_ref {
208    // The case of the default `Environment`
209    ( $trait_path:path ) => {
210        $crate::contract_ref!($trait_path, Environment)
211    };
212    // The case of the custom `Environment`
213    ( $trait_path:path, $env:ty ) => {
214        <<$crate::reflect::TraitDefinitionRegistry<$env> as $trait_path>
215                    ::__ink_TraitInfo as $crate::codegen::TraitCallForwarder>::Forwarder
216    };
217}
