15/// Trait implemented by chain extensions.
16///
17/// Allows to use the `self.env().extension().my_chain_extension(...)` syntax.
18///
19/// # Note
20///
21/// This trait is automatically implemented when using `#[ink::chain_extension]`
22/// procedural macro.
23pub trait ChainExtensionInstance {
24    /// The type of the chain extension instance.
25    type Instance;
26
27    /// Creates a new instance of the chain extension to use methods with method chaining
28    /// syntax.
29    fn instantiate() -> Self::Instance;
30}
31
32/// Implemented by chain extension types.
33///
34/// Every chain extension defines a set of chain extension methods
35/// that share a common error code type.
36#[allow(dead_code)]
37pub trait ChainExtension {
38    /// The error code that determines whether a chain extension method call was
39    /// successful.
40    type ErrorCode: ink_env::chain_extension::FromStatusCode;
41}
42
43/// Only implemented for `Result<T, E>`.
44///
45/// Used to check at compile time if the chain extension method return type
46/// is a `Result` type using the type system instead of the syntactic structure.
47#[doc(hidden)]
48pub trait IsResultType: private::IsResultSealed {
49    /// The `T` type of the `Result<T, E>`.
50    type Ok;
51    /// The `E` type of the `Result<T, E>`.
52    type Err;
53}
54
55impl<T, E> private::IsResultSealed for Result<T, E> {}
56impl<T, E> IsResultType for Result<T, E> {
57    type Ok = T;
58    type Err = E;
59}
60
61/// Only implemented for [`ValueReturned`].
62///
63/// Used to deduce the correct return type of a chain extension method at compile time
64/// based on 2 flags: `const IS_RESULT: bool` and `const HANDLE_STATUS: bool`.
65///
66/// If `IS_RESULT` is set to `false` and `HANDLE_STATUS` is `true`, then
67/// `type ReturnType = Result<T, E>`. Otherwise `type ReturnType = T`.
68pub trait Output<const IS_RESULT: bool, const HANDLE_STATUS: bool, T, E>:
69    private::OutputSealed
70{
71    type ReturnType;
72}
73
74/// Represents some abstract value that is returned by a function.
75/// Currently acts as a placeholder.
76pub struct ValueReturned;
77impl private::OutputSealed for ValueReturned {}
78
79impl<T, E> Output<false, false, T, E> for ValueReturned {
80    type ReturnType = T;
81}
82
83impl<T, E> Output<false, true, T, E> for ValueReturned {
84    type ReturnType = core::result::Result<T, E>;
85}
86
87impl<T, E> Output<true, false, T, E> for ValueReturned {
88    type ReturnType = T;
89}
90
91impl<T, E> Output<true, true, T, E> for ValueReturned {
92    type ReturnType = T;
93}

95mod private {
96    /// Seals the `IsResultType` trait so that it cannot be implemented outside this
97    /// module.
98    pub trait IsResultSealed {}
99    /// Seals the `Output` trait so that it cannot be implemented outside this module.
100    pub trait OutputSealed {}
101}

103/// Macro defines the combined chain extension via structure definition.
104/// Each sub-extension can be accessed by the corresponding field.
105///
106/// The macro expects a structure definition as an input where each field should
107/// implement [`ChainExtensionInstance`]. Usually, this trait is implemented
108/// by the `#[ink::chain_extension]` macro during the definition of the chain extension.
109///
110/// ```rust
111/// #[ink::scale_derive(TypeInfo)]
112/// struct ExtensionOne;
113/// impl ink::ChainExtensionInstance for ExtensionOne {
114///     type Instance = Self;
115///
116///     fn instantiate() -> Self::Instance {
117///         Self {}
118///     }
119/// }
120///
121/// #[ink::scale_derive(TypeInfo)]
122/// struct ExtensionTwo;
123/// impl ink::ChainExtensionInstance for ExtensionTwo {
124///     type Instance = Self;
125///
126///     fn instantiate() -> Self::Instance {
127///         Self {}
128///     }
129/// }
130///
131/// ink::combine_extensions! {
132///     /// Defines a combined extension with [`ExtensionOne`] and [`ExtensionTwo`].
133///     struct Combined {
134///         /// This field can be used to access the first extension like
135///         /// `self.env().extension().extension_one` in the contract's context.
136///         extension_one: ExtensionOne,
137///         /// This field can be used to access the second extension like
138///         /// `self.env().extension().extension_two` in the contract's context.
139///         extension_two: ExtensionTwo,
140///     }
141/// }
142/// ```
143#[macro_export]
144macro_rules! combine_extensions {
145    ($(#[$meta:meta])* $vis:vis struct $name:ident {
146        $($(#[$field_meta:meta])* $field_vis:vis $field_name:ident: $field_type:ty,)*
147    }) => {
148        $(#[$meta])*
149        #[::ink::scale_derive(TypeInfo)]
150        $vis struct $name {
151            $($(#[$field_meta])* $field_vis $field_name: $field_type,)*
152        }
153
154        const _: () = {
155            /// Each chain extension has an abstract type `$name` that describes
156            /// it and the actual instance that provides access to methods.
157            /// This structure is an instance that is returned by the `self.env().extension()` call.
158            ///
159            /// Because it is a combination of corresponding sub-instances, we need to initialize
160            /// each sub-instance in the same way by calling [`ChainExtensionInstance::instantiate`].
161            pub struct Private {
162                $($(#[$field_meta])* $field_vis $field_name: <$field_type as ::ink::ChainExtensionInstance>::Instance,)*
163            }
164
165            impl ::ink::ChainExtensionInstance for $name {
166                type Instance = Private;
167
168                fn instantiate() -> Self::Instance {
169                    Self::Instance {
170                        $($field_name: <$field_type as ::ink::ChainExtensionInstance>::instantiate(),)*
171                    }
172                }
173            }
174        };
175    }
176}
