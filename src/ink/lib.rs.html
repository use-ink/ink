119/// Any event which derives `#[derive(ink::EventMetadata)]` and is used in the contract
120/// binary will have its implementation added to this distributed slice at linking time.
121#[cfg(feature = "std")]
122#[linkme::distributed_slice]
123#[linkme(crate = linkme)]
124pub static CONTRACT_EVENTS: [fn() -> EventSpec] = [..];
125
126/// Collect the [`EventSpec`] metadata of all event definitions linked and used in the
127/// binary.
128#[cfg(feature = "std")]
129pub fn collect_events() -> Vec<EventSpec> {
130    CONTRACT_EVENTS.iter().map(|event| event()).collect()
131}
132
133/// Any event whose parameters type implement `ink::SolDecode` and `ink::SolEncode`
134/// and is used in the contract binary will have its implementation added to this
135/// distributed slice at linking time.
136#[cfg(all(feature = "std", any(ink_abi = "sol", ink_abi = "all")))]
137#[linkme::distributed_slice]
138#[linkme(crate = linkme)]
139pub static CONTRACT_EVENTS_SOL: [fn() -> ink_metadata::sol::EventMetadata] = [..];
140
141/// Collect the Solidity ABI compatible metadata of all event definitions linked and used
142/// in the binary.
143#[cfg(all(feature = "std", any(ink_abi = "sol", ink_abi = "all")))]
144pub fn collect_events_sol() -> Vec<ink_metadata::sol::EventMetadata> {
145    crate::CONTRACT_EVENTS_SOL
146        .iter()
147        .map(|event| event())
148        .collect()
149}
150
151/// Any error which derives `#[derive(ink::SolErrorMetadata)]` and is used in the contract
152/// binary will have its implementation added to this distributed slice at linking time.
153#[cfg(feature = "std")]
154#[linkme::distributed_slice]
155#[linkme(crate = linkme)]
156pub static CONTRACT_ERRORS_SOL: [fn() -> Vec<ink_metadata::sol::ErrorMetadata>] = [..];
157
158/// Collect the Solidity ABI compatible metadata of all error definitions encoded as
159/// Solidity custom errors that are linked and used in the binary.
160#[cfg(feature = "std")]
161pub fn collect_errors_sol() -> Vec<ink_metadata::sol::ErrorMetadata> {
162    crate::CONTRACT_ERRORS_SOL
163        .iter()
164        .flat_map(|event| event())
165        .collect()
166}
