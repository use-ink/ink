15/// Creates an instance of a message builder for an `#[ink::trait_definition]`.
16///
17/// This is done by creating a wrapper around the trait defined with the
18/// [`ink::trait_definition`](crate::trait_definition) macro.
19///
20/// The macro returns an instance of the generated message builder type which implements
21/// the trait, allowing the user to create and invoke messages on the trait.
22///
23/// This is similar to the call builder syntax accessible via the [`crate::contract_ref!`]
24/// macro, except that it is decoupled from the callee account id, as well as the
25/// underlying execution environment. This allows it to be used in execution contexts
26/// other than cross-contract calls.
27///
28/// # Usage
29///
30/// The macro expects two arguments:
31/// - The first argument is the path to the trait, e.g. `Erc20` or `erc20::Erc20`.
32/// - The second argument is the type of the [`ink_env::Environment`].
33///
34/// If the second argument is not specified, the macro uses the
35/// [`ink_env::DefaultEnvironment`].
36///
37/// ```rust
38/// use ink::message_builder;
39/// use ink_env::{
40///     call::{
41///         ExecutionInput,
42///         Executor,
43///     },
44///     DefaultEnvironment,
45/// };
46/// use ink_primitives::{
47///     AccountId,
48///     Address,
49///     MessageResult,
50/// };
51/// use scale::{
52///     Decode,
53///     Encode,
54/// };
55///
56/// #[ink::trait_definition]
57/// pub trait Erc20 {
58///     /// Returns the total supply of the ERC-20 smart contract.
59///     #[ink(message)]
60///     fn total_supply(&self) -> u128;
61///
62///     /// Transfers balance from the caller to the given address.
63///     #[ink(message)]
64///     fn transfer(&mut self, amount: u128, to: Address) -> bool;
65/// }
66///
67/// #[derive(Clone)]
68/// pub struct CustomEnv;
69///
70/// impl ink_env::Environment for CustomEnv {
71///     const MAX_EVENT_TOPICS: usize = 3;
72///     type AccountId = [u8; 32];
73///     type Balance = u64;
74///     type Hash = [u8; 32];
75///     type Timestamp = u64;
76///     type BlockNumber = u64;
77///     type ChainExtension = ();
78///     type EventRecord = ();
79/// }
80///
81/// /// To demonstrate implementing an execution environment agnostic executor
82/// pub struct ExampleExecutor<E> {
83///     marker: core::marker::PhantomData<E>,
84/// }
85///
86/// impl<E> ExampleExecutor<E> {
87///     pub fn new() -> Self {
88///         Self {
89///             marker: core::marker::PhantomData,
90///         }
91///     }
92/// }
93///
94/// impl<E> Executor<E> for ExampleExecutor<E>
95/// where
96///     E: ink_env::Environment,
97/// {
98///     type Error = ();
99///     fn exec<Args, Output, Abi>(
100///         &self,
101///         input: &ExecutionInput<Args, Abi>,
102///     ) -> Result<MessageResult<Output>, Self::Error>
103///     where
104///         Args: ink::reflect::AbiEncodeWith<Abi>,
105///         Output: ink::reflect::AbiDecodeWith<Abi>,
106///     {
107///         println!("Executing contract with input: {:?}", input.encode());
108///         unimplemented!("Decode contract execution output")
109///     }
110/// }
111///
112/// fn default(to: Address) {
113///     let executor = ExampleExecutor::<DefaultEnvironment>::new();
114///     let mut contract = message_builder!(Erc20);
115///     let total_supply = contract.total_supply().exec(&executor).unwrap().unwrap();
116///     contract.transfer(total_supply, to).exec(&executor).unwrap();
117/// }
118///
119/// fn custom(to: Address) {
120///     let executor = ExampleExecutor::<CustomEnv>::new();
121///     let mut contract = message_builder!(Erc20, CustomEnv);
122///     let total_supply = contract.total_supply().exec(&executor).unwrap().unwrap();
123///     contract.transfer(total_supply, to).exec(&executor).unwrap();
124/// }
125///
126/// fn generic<E>(to: Address)
127/// where
128///     E: ink_env::Environment,
129/// {
130///     let executor = ExampleExecutor::<E>::new();
131///     let mut contract = message_builder!(Erc20, E);
132///     let total_supply = contract.total_supply().exec(&executor).unwrap().unwrap();
133///     contract.transfer(total_supply, to).exec(&executor).unwrap();
134/// }
135/// ```
136#[macro_export]
137macro_rules! message_builder {
138    // The case of the default `Environment`
139    ( $trait_path:path ) => {
140        $crate::message_builder!($trait_path, $crate::env::DefaultEnvironment)
141    };
142    // The case of the custom `Environment`
143    ( $trait_path:path, $env:ty ) => {
144        <<<$crate::reflect::TraitDefinitionRegistry<$env>
145                            as $trait_path>::__ink_TraitInfo
146                            as $crate::codegen::TraitMessageBuilder>::MessageBuilder
147                            as ::core::default::Default>::default()
148    };
149}
