15/// Creates an instance of a message builder for an `#[ink::trait_definition]`.
16///
17/// This is done by creating a wrapper around the trait defined with the
18/// [`ink::trait_definition`](crate::trait_definition) macro.
19///
20/// The macro returns an instance of the generated message builder type which implements
21/// the trait, allowing the user to create and invoke messages on the trait.
22///
23/// This is similar to the call builder syntax accessible via the [`crate::contract_ref!`]
24/// macro, except that it is decoupled from the callee account id, as well as the
25/// underlying execution environment. This allows it to be used in execution contexts
26/// other than cross-contract calls.
27///
28/// # Usage
29///
30/// The macro expects up to three arguments:
31/// - The first argument is the path to the trait, e.g. `Erc20` or `erc20::Erc20`.
32/// - The second argument is the type of the [`ink_env::Environment`].
33/// - The third argument is the marker type for the ABI (i.e.
34///   [`ink::abi::Ink`][crate::abi::Ink] or [`ink::abi::Sol`][crate::abi::Sol]).
35///
36/// If the second argument is not specified, the macro uses the
37/// [`ink_env::DefaultEnvironment`].
38/// If the third argument is not specified, the macro uses the "default" ABI for calls
39/// for the ink! project.
40///
41/// # Note
42///
43/// The "default" ABI for calls is "ink", unless the ABI is set to "sol"
44/// in the ink! project's manifest file (i.e. `Cargo.toml`).
45///
46/// ```rust
47/// use ink::message_builder;
48/// use ink_env::{
49///     call::{
50///         ExecutionInput,
51///         Executor,
52///     },
53///     DefaultEnvironment,
54/// };
55/// use ink_primitives::{
56///     AccountId,
57///     Address,
58///     MessageResult,
59/// };
60/// use scale::{
61///     Decode,
62///     Encode,
63/// };
64///
65/// #[ink::trait_definition]
66/// pub trait Erc20 {
67///     /// Returns the total supply of the ERC-20 smart contract.
68///     #[ink(message)]
69///     fn total_supply(&self) -> u128;
70///
71///     /// Transfers balance from the caller to the given address.
72///     #[ink(message)]
73///     fn transfer(&mut self, amount: u128, to: Address) -> bool;
74/// }
75///
76/// #[derive(Clone)]
77/// pub struct CustomEnv;
78///
79/// impl ink_env::Environment for CustomEnv {
80///     const MAX_EVENT_TOPICS: usize = 3;
81///     const NATIVE_TO_ETH_RATIO: u32 = 100_000_000;
82///     type AccountId = [u8; 32];
83///     type Balance = u64;
84///     type Hash = [u8; 32];
85///     type Timestamp = u64;
86///     type BlockNumber = u64;
87///     type ChainExtension = ();
88///     type EventRecord = ();
89/// }
90///
91/// /// To demonstrate implementing an execution environment agnostic executor
92/// pub struct ExampleExecutor<E> {
93///     marker: core::marker::PhantomData<E>,
94/// }
95///
96/// impl<E> ExampleExecutor<E> {
97///     pub fn new() -> Self {
98///         Self {
99///             marker: core::marker::PhantomData,
100///         }
101///     }
102/// }
103///
104/// impl<E> Executor<E> for ExampleExecutor<E>
105/// where
106///     E: ink_env::Environment,
107/// {
108///     type Error = ();
109///     fn exec<Args, Output, Abi>(
110///         &self,
111///         input: &ExecutionInput<Args, Abi>,
112///     ) -> Result<MessageResult<Output>, Self::Error>
113///     where
114///         Args: ink::abi::AbiEncodeWith<Abi>,
115///         Output: ink::env::call::utils::DecodeMessageResult<Abi>,
116///     {
117///         println!("Executing contract with input: {:?}", input.encode());
118///         unimplemented!("Decode contract execution output")
119///     }
120/// }
121///
122/// fn default(to: Address) {
123///     let executor = ExampleExecutor::<DefaultEnvironment>::new();
124///     let mut contract = message_builder!(Erc20);
125///     let total_supply = contract.total_supply().exec(&executor).unwrap().unwrap();
126///     contract.transfer(total_supply, to).exec(&executor).unwrap();
127/// }
128///
129/// fn custom(to: Address) {
130///     let executor = ExampleExecutor::<CustomEnv>::new();
131///     let mut contract = message_builder!(Erc20, CustomEnv);
132///     let total_supply = contract.total_supply().exec(&executor).unwrap().unwrap();
133///     contract.transfer(total_supply, to).exec(&executor).unwrap();
134/// }
135///
136/// fn custom_abi(to: Address) {
137///     let executor = ExampleExecutor::<DefaultEnvironment>::new();
138///     let mut contract = message_builder!(Erc20, DefaultEnvironment, ink::abi::Ink);
139///     let total_supply = contract.total_supply().exec(&executor).unwrap().unwrap();
140///     contract.transfer(total_supply, to).exec(&executor).unwrap();
141/// }
142///
143/// fn generic<E>(to: Address)
144/// where
145///     E: ink_env::Environment,
146/// {
147///     let executor = ExampleExecutor::<E>::new();
148///     let mut contract = message_builder!(Erc20, E);
149///     let total_supply = contract.total_supply().exec(&executor).unwrap().unwrap();
150///     contract.transfer(total_supply, to).exec(&executor).unwrap();
151/// }
152/// ```
153#[macro_export]
154macro_rules! message_builder {
155    // The case of the default `Environment` and ABI
156    ( $trait_path:path ) => {
157        $crate::message_builder!($trait_path, $crate::env::DefaultEnvironment)
158    };
159    // The case of the custom `Environment` and default ABI
160    ( $trait_path:path, $env:ty ) => {
161        $crate::message_builder!($trait_path, $env, $crate::env::DefaultAbi)
162    };
163    // The case of the custom `Environment` and ABI
164    ( $trait_path:path, $env:ty, $abi:ty ) => {
165        <<<$crate::reflect::TraitDefinitionRegistry<$env>
166                            as $trait_path>::__ink_TraitInfo
167                            as $crate::codegen::TraitMessageBuilder>::MessageBuilder<$abi>
168                            as ::core::default::Default>::default()
169    };
170}
