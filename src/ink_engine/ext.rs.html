36/// The off-chain engine.
37pub struct Engine {
38    /// The environment database.
39    pub database: Database,
40    /// The current execution context.
41    pub exec_context: ExecContext,
42    /// Recorder for relevant interactions with the engine.
43    /// This is specifically about debug info. This info is
44    /// not available in the `contracts` pallet.
45    pub(crate) debug_info: DebugInfo,
46    /// The chain specification.
47    pub chain_spec: ChainSpec,
48}
49
50/// The chain specification.
51pub struct ChainSpec {
52    /// The current gas price.
53    pub gas_price: U256,
54    /// The minimum value an account of the chain must have
55    /// (i.e. the chain's existential deposit).
56    pub minimum_balance: U256,
57    /// The targeted block time.
58    pub block_time: BlockTimestamp,
59}
60
61/// The default values for the chain specification are:
62///
63///   * `gas_price`: 100
64///   * `minimum_balance`: 42
65///   * `block_time`: 6
66///
67/// There is no particular reason behind choosing them this way.
68impl Default for ChainSpec {
69    fn default() -> Self {
70        Self {
71            gas_price: 100.into(),
72            minimum_balance: 42.into(),
73            block_time: 6,
74        }
75    }
76}

96impl Engine {
97    /// Transfers value from the contract to the destination account.
98    #[allow(clippy::arithmetic_side_effects)] // todo
99    pub fn transfer(&mut self, dest: Address, mut value: &[u8]) -> Result<(), Error> {
100        // Note that a transfer of `0` is allowed here
101        let increment = <u128 as scale::Decode>::decode(&mut value)
102            .map_err(|_| Error::TransferFailed)?;
103
104        // Note that the destination account does not have to exist
105        let dest_old_balance = self.get_balance(dest).unwrap_or_default();
106
107        let contract = self.get_callee();
108        let contract_old_balance = self
109            .get_balance(contract)
110            .map_err(|_| Error::TransferFailed)?;
111
112        self.database
113            .set_balance(&contract, contract_old_balance - increment);
114        self.database
115            .set_balance(&dest, dest_old_balance + increment);
116        Ok(())
117    }

119    /// Deposits an event identified by the supplied topics and data.
120    pub fn deposit_event(&mut self, topics: &[[u8; 32]], data: &[u8]) {
121        self.debug_info.record_event(EmittedEvent {
122            topics: topics.to_vec(),
123            data: data.to_vec(),
124        });
125    }

127    /// Writes the encoded value into the storage at the given key.
128    /// Returns the size of the previously stored value at the key if any.
129    pub fn set_storage(&mut self, key: &[u8], encoded_value: &[u8]) -> Option<u32> {
130        let callee = self.get_callee();
131
132        self.debug_info.inc_writes(callee);
133        self.debug_info
134            .record_cell_for_account(callee, key.to_vec());
135
136        self.database
137            .insert_into_contract_storage(&callee, key, encoded_value.to_vec())
138            .map(|v| <u32>::try_from(v.len()).expect("usize to u32 conversion failed"))
139    }
140
141    /// Returns the contract storage bytes at the key if any.
142    pub fn get_storage(&mut self, key: &[u8]) -> Result<&[u8], Error> {
143        let callee = self.get_callee();
144
145        self.debug_info.inc_reads(callee);
146        match self.database.get_from_contract_storage(&callee, key) {
147            Some(val) => Ok(val),
148            None => Err(Error::KeyNotFound),
149        }
150    }
151
152    /// Removes the storage entries at the given key,
153    /// returning previously stored value at the key if any.
154    pub fn take_storage(&mut self, key: &[u8]) -> Result<Vec<u8>, Error> {
155        let callee = self.get_callee();
156
157        self.debug_info.inc_writes(callee);
158        match self.database.remove_contract_storage(&callee, key) {
159            Some(val) => Ok(val),
160            None => Err(Error::KeyNotFound),
161        }
162    }
163
164    /// Returns the size of the value stored in the contract storage at the key if any.
165    pub fn contains_storage(&mut self, key: &[u8]) -> Option<u32> {
166        let callee = self.get_callee();
167
168        self.debug_info.inc_reads(callee);
169        self.database
170            .get_from_contract_storage(&callee, key)
171            .map(|val| val.len() as u32)
172    }
173
174    /// Removes the storage entries at the given key.
175    /// Returns the size of the previously stored value at the key if any.
176    pub fn clear_storage(&mut self, key: &[u8]) -> Option<u32> {
177        let callee = self.get_callee();
178        self.debug_info.inc_writes(callee);
179        let _ = self
180            .debug_info
181            .remove_cell_for_account(callee, key.to_vec());
182        self.database
183            .remove_contract_storage(&callee, key)
184            .map(|val| val.len() as u32)
185    }
186
187    /// Remove the calling account and transfer remaining balance.
188    ///
189    /// todo is the following comment still correct?
190    /// This function never returns. Either the termination was successful and the
191    /// execution of the destroyed contract is halted. Or it failed during the
192    /// termination which is considered fatal.
193    pub fn terminate(&mut self, beneficiary: Address) -> Result<(), Error> {
194        // Send the remaining balance to the beneficiary
195        let contract = self.get_callee();
196        let all = self
197            .get_balance(contract)
198            .unwrap_or_else(|err| panic!("could not get balance: {err:?}"));
199        let value = &scale::Encode::encode(&all)[..];
200        self.transfer(beneficiary, value)
201            .unwrap_or_else(|err| panic!("transfer did not work: {err:?}"));
202
203        Ok(())
204    }

320
321    pub fn call(
322        &mut self,
323        callee: &[u8],
324        _gas_limit: u64,
325        _value: &[u8],
326        input: &[u8],
327        output: &mut &mut [u8],
328    ) -> Result<(), Error> {
329        const ECRECOVER: [u8; 20] = hex!("0000000000000000000000000000000000000001");
330        if callee == ECRECOVER {
331            let mut signature = [0u8; 65];
332            signature.copy_from_slice(&input[..65]);
333            let mut message_hash = [0u8; 32];
334            message_hash.copy_from_slice(&input[65..65 + 32]);
335
336            let out: &mut [u8; 33] = output
337                .as_mut()
338                .try_into()
339                .expect("Slice must be exactly 33 bytes long");
340            let _ = self.ecdsa_recover(&signature, &message_hash, out);
341        }
342        unimplemented!(
343            "off-chain environment does not yet support `call` for non-precompiles"
344        );
345    }

347    /// Emulates gas price calculation.
348    pub fn weight_to_fee(&self, gas: u64, output: &mut &mut [u8]) {
349        let fee = self.chain_spec.gas_price.saturating_mul(gas.into());
350        let fee: Vec<u8> = scale::Encode::encode(&fee);
351        set_output(output, &fee[..])
352    }

355impl Engine {
356    /// Recovers the compressed ECDSA public key for given `signature` and `message_hash`,
357    /// and stores the result in `output`.
358    #[allow(clippy::arithmetic_side_effects)] // todo
359    pub fn ecdsa_recover(
360        &mut self,
361        signature: &[u8; 65],
362        message_hash: &[u8; 32],
363        output: &mut [u8; 33],
364    ) -> Result<(), Error> {
365        use secp256k1::{
366            Message,
367            SECP256K1,
368            ecdsa::{
369                RecoverableSignature,
370                RecoveryId,
371            },
372        };
373
374        // In most implementations, the v is just 0 or 1 internally, but 27 was added
375        // as an arbitrary number for signing Bitcoin messages and Ethereum adopted that
376        // as well.
377        let recovery_byte = if signature[64] > 26 {
378            signature[64] - 27
379        } else {
380            signature[64]
381        };
382
383        let recovery_id = RecoveryId::try_from(recovery_byte as i32)
384            .unwrap_or_else(|error| panic!("Unable to parse the recovery id: {error}"));
385
386        let message = Message::from_digest_slice(message_hash).unwrap_or_else(|error| {
387            panic!("Unable to create the message from hash: {error}")
388        });
389        let signature =
390            RecoverableSignature::from_compact(&signature[0..64], recovery_id)
391                .unwrap_or_else(|error| panic!("Unable to parse the signature: {error}"));
392
393        let pub_key = SECP256K1.recover_ecdsa(&message, &signature);
394        match pub_key {
395            Ok(pub_key) => {
396                *output = pub_key.serialize();
397                Ok(())
398            }
399            Err(_) => Err(Error::EcdsaRecoveryFailed),
400        }
401    }
402}

404/// Copies the `slice` into `output`.
405///
406/// Panics if the slice is too large and does not fit.
407fn set_output(output: &mut &mut [u8], slice: &[u8]) {
408    assert!(
409        slice.len() <= output.len(),
410        "the output buffer is too small! the decoded storage is of size {} bytes, \
411        but the output buffer has only room for {}.",
412        slice.len(),
413        output.len(),
414    );
415    output[..slice.len()].copy_from_slice(slice);
416}
