44/// A smart contract execution call trace.
45#[derive(
46    TypeInfo, Default, Encode, Decode, Serialize, Deserialize, Clone, Debug, Eq, PartialEq,
47)]
48#[serde(rename_all = "camelCase")]
49pub struct CallTrace<Gas = U256> {
50    /// Address of the sender.
51    pub from: H160,
52    /// Amount of gas provided for the call.
53    pub gas: Gas,
54    /// Amount of gas used.
55    pub gas_used: Gas,
56    /// Address of the receiver.
57    pub to: H160,
58    /// Call input data.
59    pub input: Bytes,
60    /// Return data.
61    #[serde(skip_serializing_if = "Bytes::is_empty")]
62    pub output: Bytes,
63    /// The error message if the call failed.
64    #[serde(skip_serializing_if = "Option::is_none")]
65    pub error: Option<String>,
66    /// The revert reason, if the call reverted.
67    #[serde(skip_serializing_if = "Option::is_none")]
68    pub revert_reason: Option<String>,
69    /// List of sub-calls.
70    #[serde(skip_serializing_if = "Vec::is_empty")]
71    pub calls: Vec<CallTrace<Gas>>,
72    /// List of logs emitted during the call.
73    #[serde(skip_serializing_if = "Vec::is_empty")]
74    pub logs: Vec<CallLog>,
75    /// Amount of value transferred.
76    #[serde(skip_serializing_if = "Option::is_none")]
77    pub value: Option<U256>,
78    /// Type of call.
79    #[serde(rename = "type")]
80    pub call_type: CallType,
81    /// Number of child calls entered (for log position calculation)
82    #[serde(skip)]
83    pub child_call_count: u32,
84}

86/// A log emitted during a call.
87#[derive(
88    Debug, Default, Clone, Encode, Decode, TypeInfo, Serialize, Deserialize, Eq, PartialEq,
89)]
90pub struct CallLog {
91    /// The address of the contract that emitted the log.
92    pub address: H160,
93    /// The topics used to index the log.
94    #[serde(default, skip_serializing_if = "Vec::is_empty")]
95    pub topics: Vec<H256>,
96    /// The log's data.
97    pub data: Vec<u8>,
98    /// Position of the log relative to subcalls within the same trace
99    /// See <https://github.com/ethereum/go-ethereum/pull/28389> for details
100    #[serde(with = "super::hex_serde")]
101    pub position: u32,
102}

157/// A prestate Trace
158#[derive(
159    TypeInfo, Encode, Decode, Serialize, Deserialize, Clone, Debug, Eq, PartialEq,
160)]
161#[serde(untagged)]
162pub enum PrestateTrace {
163    /// The Prestate mode returns the accounts necessary to execute a given transaction
164    Prestate(BTreeMap<H160, PrestateTraceInfo>),
165
166    /// The diff mode returns the differences between the transaction's pre and post-state
167    /// The result only contains the accounts that were modified by the transaction
168    DiffMode {
169        /// The state before the call.
170        ///  The accounts in the `pre` field will contain all of their basic fields, even
171        /// if those fields have not been modified. For `storage` however, only
172        /// non-empty slots that have been modified will be included
173        pre: BTreeMap<H160, PrestateTraceInfo>,
174        /// The state after the call.
175        /// It only contains the specific fields that were actually modified during the
176        /// transaction
177        post: BTreeMap<H160, PrestateTraceInfo>,
178    },
179}
180
181impl PrestateTrace {
182    /// Returns the pre and post trace info.
183    pub fn state_mut(
184        &mut self,
185    ) -> (
186        &mut BTreeMap<H160, PrestateTraceInfo>,
187        Option<&mut BTreeMap<H160, PrestateTraceInfo>>,
188    ) {
189        match self {
190            PrestateTrace::Prestate(pre) => (pre, None),
191            PrestateTrace::DiffMode { pre, post } => (pre, Some(post)),
192        }
193    }
194}

196/// The info of a prestate trace.
197#[derive(
198    TypeInfo, Default, Encode, Decode, Serialize, Deserialize, Clone, Debug, Eq, PartialEq,
199)]
200pub struct PrestateTraceInfo {
201    /// The balance of the account.
202    #[serde(skip_serializing_if = "Option::is_none")]
203    pub balance: Option<U256>,
204    /// The nonce of the account.
205    #[serde(skip_serializing_if = "Option::is_none")]

208    #[serde(skip_serializing_if = "Option::is_none")]

211    #[serde(

213        serialize_with = "serialize_map_skip_none"
214    )]
215    pub storage: BTreeMap<Vec<u8>, Option<Vec<u8>>>,
216}

218/// Returns true if the map has no `Some` element
219pub fn is_empty<K, V>(map: &BTreeMap<K, Option<V>>) -> bool {
220    !map.values().any(|v| v.is_some())
221}
222
223/// Serializes a map, skipping `None` values.
224pub fn serialize_map_skip_none<S, K, V>(
225    map: &BTreeMap<K, Option<V>>,
226    serializer: S,
227) -> Result<S::Ok, S::Error>
228where
229    S: Serializer,
230    K: serde::Serialize,
231    V: serde::Serialize,
232{
233    let len = map.values().filter(|v| v.is_some()).count();
234    let mut ser_map = serializer.serialize_map(Some(len))?;
235
236    for (key, opt_val) in map {
237        if let Some(val) = opt_val {
238            ser_map.serialize_entry(key, val)?;
239        }
240    }
241
242    ser_map.end()
243}

245/// The type of tracer to use.
246/// Only "callTracer" is supported for now.
247#[derive(TypeInfo, Debug, Clone, Encode, Decode, Serialize, Deserialize, PartialEq)]
248#[serde(tag = "tracer", content = "tracerConfig", rename_all = "camelCase")]
249pub enum TracerType {
250    /// A tracer that traces calls.
251    CallTracer(Option<CallTracerConfig>),
252
253    /// A tracer that traces the prestate.
254    PrestateTracer(Option<PrestateTracerConfig>),
255}
256
257impl From<CallTracerConfig> for TracerType {
258    fn from(config: CallTracerConfig) -> Self {
259        TracerType::CallTracer(Some(config))
260    }
261}
262
263impl Default for TracerType {
264    fn default() -> Self {
265        TracerType::CallTracer(Some(CallTracerConfig::default()))
266    }
267}
