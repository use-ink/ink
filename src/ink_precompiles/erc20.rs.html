15//! ERC-20 precompile interface for `pallet-assets`.
16//!
17//! This module provides the standard ERC-20 token interface for interacting with
18//! assets managed by `pallet-assets`.
19//!
20//! # References
21//!
22//! - [Polkadot SDK Assets Precompile](https://github.com/paritytech/polkadot-sdk/blob/master/substrate/frame/assets/precompiles/src/lib.rs)
23//! - [Polkadot SDK Assets Precompile Solidity Interface](https://github.com/paritytech/polkadot-sdk/blob/master/substrate/primitives/ethereum-standards/src/IERC20.sol)
24//! - [ERC-20 Token Standard](https://eips.ethereum.org/EIPS/eip-20)
25
26use ink::{
27    Address,
28    U256,
29};
30
31/// Type alias for asset IDs.
32pub type AssetId = u32;
33
34/// Defines the ERC-20 interface of the Asset Hub precompile.
35#[ink::contract_ref(abi = "sol")]
36pub trait Erc20 {
37    /// Returns the total supply of tokens.
38    ///
39    /// # Solidity Signature
40    ///
41    /// ```solidity
42    /// function totalSupply() external view returns (uint256);
43    /// ```
44    #[ink(message)]
45    #[allow(non_snake_case)]
46    fn totalSupply(&self) -> U256;
47
48    /// Returns the balance of an account.
49    ///
50    /// # Arguments
51    /// * `account` - The address to query the balance of
52    ///
53    /// # Solidity Signature
54    ///
55    /// ```solidity
56    /// function balanceOf(address account) external view returns (uint256);
57    /// ```
58    #[ink(message)]
59    #[allow(non_snake_case)]
60    fn balanceOf(&self, account: Address) -> U256;
61
62    /// Transfers tokens to another account.
63    ///
64    /// # Arguments
65    /// * `to` - The recipient address
66    /// * `value` - The amount of tokens to transfer
67    ///
68    /// # Returns
69    ///
70    /// Returns `true` if the transfer was successful.
71    ///
72    /// # Solidity Signature
73    ///
74    /// ```solidity
75    /// function transfer(address to, uint256 value) external returns (bool);
76    /// ```
77    #[ink(message)]
78    fn transfer(&mut self, to: Address, value: U256) -> bool;
79
80    /// Returns the allowance for a spender on behalf of an owner.
81    ///
82    /// This shows how many tokens `spender` is allowed to spend on behalf of `owner`.
83    ///
84    /// # Arguments
85    /// * `owner` - The token owner's address
86    /// * `spender` - The spender's address
87    ///
88    /// # Solidity Signature
89    ///
90    /// ```solidity
91    /// function allowance(address owner, address spender) external view returns (uint256);
92    /// ```
93    #[ink(message)]
94    fn allowance(&self, owner: Address, spender: Address) -> U256;
95
96    /// Approves a spender to spend tokens on behalf of the caller.
97    ///
98    /// # Arguments
99    /// * `spender` - The address authorized to spend tokens
100    /// * `value` - The maximum amount the spender can spend
101    ///
102    /// # Returns
103    ///
104    /// Returns `true` if the approval was successful.
105    ///
106    /// # Solidity Signature
107    ///
108    /// ```solidity
109    /// function approve(address spender, uint256 value) external returns (bool);
110    /// ```
111    #[ink(message)]
112    fn approve(&mut self, spender: Address, value: U256) -> bool;
113
114    /// Transfers tokens from one account to another using allowance.
115    ///
116    /// The caller must have sufficient allowance from the `from` account.
117    ///
118    /// # Arguments
119    /// * `from` - The address to transfer tokens from
120    /// * `to` - The recipient address
121    /// * `value` - The amount of tokens to transfer
122    ///
123    /// # Returns
124    ///
125    /// Returns `true` if the transfer was successful.
126    ///
127    /// # Solidity Signature
128    ///
129    /// ```solidity
130    /// function transferFrom(address from, address to, uint256 value) external returns (bool);
131    /// ```
132    #[ink(message)]
133    #[allow(non_snake_case)]
134    fn transferFrom(&mut self, from: Address, to: Address, value: U256) -> bool;
135}
136
137/// Creates a new ERC-20 precompile reference for the given asset ID.
138///
139/// # Arguments
140/// * `asset_id` - The ID of the asset to interact with
141///
142/// # Returns
143///
144/// Returns an `Erc20Ref` that can be used to call precompile methods.
145///
146/// # Example
147///
148/// ```rust,ignore
149/// use ink_precompiles::erc20::erc20;
150///
151/// let asset_id = 1;
152/// let erc20_ref = erc20(asset_id);
153/// let balance = erc20_ref.balanceOf(account);
154/// ```
155pub fn erc20(precompile_index: u16, asset_id: AssetId) -> Erc20Ref {
156    let address = crate::prefixed_address(precompile_index, asset_id);
157    address.into()
158}

160#[cfg(test)]
161mod tests {
162    use ink::env::Environment;
163
164    #[test]
165    fn erc20_precompile_address_format() {
166        // ERC20 Assets precompile for asset ID 1 should be at the correct address
167        let expected = [
168            0x00, 0x00, 0x00, 0x01, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
169            0x00, 0x00, 0x00, 0x01, 0x20, 0x00, 0x00,
170        ];
171
172        let address = crate::prefixed_address(
173            ink::env::DefaultEnvironment::TRUST_BACKED_ASSETS_PRECOMPILE_INDEX,
174            1,
175        );
176        let address_bytes: [u8; 20] = address.into();
177
178        assert_eq!(address_bytes, expected);
179    }
180
181    #[test]
182    fn erc20_precompile_address_for_multiple_assets() {
183        // Test asset ID 42
184        let address_42 = crate::prefixed_address(
185            ink::env::DefaultEnvironment::TRUST_BACKED_ASSETS_PRECOMPILE_INDEX,
186            42,
187        );
188        let bytes_42: [u8; 20] = address_42.into();
189
190        // First 4 bytes should be asset ID (42 = 0x0000002a)
191        assert_eq!(&bytes_42[0..4], &[0x00, 0x00, 0x00, 0x2a]);
192
193        // Bytes 16-19 should be precompile index (0x0120)
194        assert_eq!(&bytes_42[16..20], &[0x01, 0x20, 0x00, 0x00]);
195    }
196}
